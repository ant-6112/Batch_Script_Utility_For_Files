# List files larger than 1 MB in the current directory and subdirectories

Get-ChildItem -Path . -Recurse -File | 
  Where-Object { $_.Length -ge 1MB } | 
  Select-Object FullName, @{Name="Size (MB)"; Expression={ $_.Length / 1MB }}

# Explanation:
# Get-ChildItem -Path . -Recurse -File: This line retrieves information about all files (excluding directories) 
#                                       in the current directory and its subdirectories.
# Where-Object { $_.Length -ge 1MB }: This line filters the result to only include files with a size greater than or equal to 1 MB (1MB is defined).
# Select-Object FullName, @{Name="Size (MB)"; Expression={ $_.Length / 1MB }}: This line selects the file's full path name and calculates the size in MB for display.
